using UnityEngine;

public class CoinManager : MonoBehaviour
{
    public static CoinManager Instance;
    public int coinCount = 5; // ‡πÄ‡∏á‡∏¥‡∏ô‡πÄ‡∏£‡∏¥‡πà‡∏°‡∏ï‡πâ‡∏ô

    // Event ‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö‡∏ï‡∏∞‡πÇ‡∏Å‡∏ô‡∏ö‡∏≠‡∏Å‡πÄ‡∏°‡∏∑‡πà‡∏≠‡πÄ‡∏á‡∏¥‡∏ô‡πÄ‡∏õ‡∏•‡∏µ‡πà‡∏¢‡∏ô (Observer Pattern)
    public event Action<int> OnCoinChanged; 

    private void Awake()
    {
        if (Instance == null) Instance = this;
        else Destroy(gameObject);
    }

    private void Start()
    {
        // ‡πÅ‡∏à‡πâ‡∏á‡πÄ‡∏ï‡∏∑‡∏≠‡∏ô‡∏Ñ‡πà‡∏≤‡πÄ‡∏£‡∏¥‡πà‡∏°‡∏ï‡πâ‡∏ô‡∏ï‡∏≠‡∏ô‡πÄ‡∏Å‡∏°‡πÄ‡∏£‡∏¥‡πà‡∏°
        OnCoinChanged?.Invoke(coinCount);
    }

    public void AddCoins(int amount)
    {
        coinCount += amount;
        OnCoinChanged?.Invoke(coinCount); // ‡πÅ‡∏à‡πâ‡∏á UI ‡πÉ‡∏´‡πâ‡∏ó‡∏≥‡∏á‡∏≤‡∏ô
    }

    public bool SpendCoins(int amount)
    {
        if (coinCount >= amount)
        {
            coinCount -= amount;
            OnCoinChanged?.Invoke(coinCount); // ‡πÅ‡∏à‡πâ‡∏á UI ‡πÉ‡∏´‡πâ‡∏ó‡∏≥‡∏á‡∏≤‡∏ô
            return true;
        }
        return false;
    }
}

using TMPro; // ‡∏à‡∏≥‡πÄ‡∏õ‡πá‡∏ô‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö TextMeshPro

public class UIController : MonoBehaviour
{
    [Header("Displays")]
    public TextMeshProUGUI coinText; // ‡∏•‡∏≤‡∏Å TextMeshPro ‡∏°‡∏≤‡πÉ‡∏™‡πà‡∏ï‡∏£‡∏á‡∏ô‡∏µ‡πâ‡πÉ‡∏ô Inspector

    private void Start()
    {
        // ‡∏•‡∏á‡∏ó‡∏∞‡πÄ‡∏ö‡∏µ‡∏¢‡∏ô‡∏ü‡∏±‡∏á‡∏Ç‡πà‡∏≤‡∏ß‡∏à‡∏≤‡∏Å CoinManager (Subscribe)
        if (CoinManager.Instance != null)

            // ‡∏≠‡∏±‡∏õ‡πÄ‡∏î‡∏ï‡∏ó‡∏±‡∏ô‡∏ó‡∏µ‡∏´‡∏ô‡∏∂‡πà‡∏á‡∏Ñ‡∏£‡∏±‡πâ‡∏á‡πÄ‡∏ú‡∏∑‡πà‡∏≠‡∏Ñ‡πà‡∏≤‡πÄ‡∏û‡∏µ‡πâ‡∏¢‡∏ô
            UpdateCoinDisplay(CoinManager.Instance.coinCount);

    private void OnDestroy()
    {
        // ‡∏¢‡∏Å‡πÄ‡∏•‡∏¥‡∏Å‡∏Å‡∏≤‡∏£‡∏ü‡∏±‡∏á‡πÄ‡∏°‡∏∑‡πà‡∏≠ UI ‡∏ñ‡∏π‡∏Å‡∏ó‡∏≥‡∏•‡∏≤‡∏¢ (Unsubscribe)

    private void UpdateCoinDisplay(int amount)
    {
        if (coinText != null)
        {
            coinText.text = amount.ToString();
        }
    }
}

public class PlantFactory : MonoBehaviour
{
    [System.Serializable]
    public struct PlantData
    {
        public string name;           // ‡∏ä‡∏∑‡πà‡∏≠ (‡∏ï‡∏±‡πâ‡∏á‡πÑ‡∏ß‡πâ‡∏Å‡∏±‡∏ô‡∏á‡∏á)
        public PlantType type;        // ‡∏õ‡∏£‡∏∞‡πÄ‡∏†‡∏ó (Enum)
        public GameObject prefab;     // ‡∏ï‡∏±‡∏ß‡πÇ‡∏°‡πÄ‡∏î‡∏•‡∏ó‡∏µ‡πà‡∏à‡∏∞‡πÄ‡∏™‡∏Å
        public float baseScale;       // ‡∏Ç‡∏ô‡∏≤‡∏î‡πÄ‡∏£‡∏¥‡πà‡∏°‡∏ï‡πâ‡∏ô (‡πÄ‡∏ä‡πà‡∏ô 0.39...)
        public int cost;              // ‡∏£‡∏≤‡∏Ñ‡∏≤‡∏Ç‡∏≤‡∏¢
    }

    // List ‡∏ô‡∏µ‡πâ‡∏à‡∏∞‡πÇ‡∏ú‡∏•‡πà‡πÉ‡∏ô Inspector ‡πÉ‡∏´‡πâ‡∏Ñ‡∏∏‡∏ì‡∏Å‡∏î + ‡πÄ‡∏û‡∏¥‡πà‡∏°‡∏ï‡πâ‡∏ô‡πÑ‡∏°‡πâ‡πÑ‡∏î‡πâ‡∏ï‡∏≤‡∏°‡πÉ‡∏à‡∏ä‡∏≠‡∏ö
    public List<PlantData> plantCatalog; 

    // ‡∏ü‡∏±‡∏á‡∏Å‡πå‡∏ä‡∏±‡∏ô‡∏ú‡∏•‡∏¥‡∏ï‡∏ï‡πâ‡∏ô‡πÑ‡∏°‡πâ (Factory Method)
    public GameObject CreatePlant(PlantType type, Vector3 position)
    {
        // ‡∏Ñ‡πâ‡∏ô‡∏´‡∏≤‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡∏à‡∏≤‡∏Å List
        PlantData data = plantCatalog.Find(p => p.type == type);

        if (data.prefab == null) 
        {
            Debug.LogError("‡πÑ‡∏°‡πà‡∏û‡∏ö‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡∏ï‡πâ‡∏ô‡πÑ‡∏°‡πâ: " + type);
            return null;
        }

        // ‡∏™‡∏£‡πâ‡∏≤‡∏á Object
        GameObject newPlant = Instantiate(data.prefab, position, Quaternion.identity);
        
        // ‡∏ï‡∏±‡πâ‡∏á‡∏Ñ‡πà‡∏≤‡∏Ç‡∏ô‡∏≤‡∏î‡πÉ‡∏´‡πâ‡∏ñ‡∏π‡∏Å‡∏ï‡πâ‡∏≠‡∏á‡∏ï‡∏≤‡∏°‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•
        Vector3 finalScale = new Vector3(data.baseScale, data.baseScale, data.baseScale);
        newPlant.transform.localScale = finalScale;

        // ‡∏ï‡∏±‡πâ‡∏á‡∏Ñ‡πà‡∏≤ Script ‡∏õ‡∏£‡∏±‡∏ö‡∏Ç‡∏ô‡∏≤‡∏î (‡∏ñ‡πâ‡∏≤‡∏°‡∏µ)
        PlantSizeAdjust plantScript = newPlant.GetComponent<PlantSizeAdjust>();
        if (plantScript != null)
        {
            plantScript.SetOriginalScale(finalScale);
        }

        return newPlant;
    }

    // ‡∏ü‡∏±‡∏á‡∏Å‡πå‡∏ä‡∏±‡∏ô‡πÄ‡∏ä‡πá‡∏Ñ‡∏£‡∏≤‡∏Ñ‡∏≤
    public int GetPlantCost(PlantType type)
    {
        PlantData data = plantCatalog.Find(p => p.type == type);
        return data.cost;
    }
}

public class PlantSpawner : MonoBehaviour
{

    public PlantFactory factory;    // ‡∏•‡∏≤‡∏Å GameObject ‡∏ó‡∏µ‡πà‡∏°‡∏µ PlantFactory ‡∏°‡∏≤‡πÉ‡∏™‡πà
